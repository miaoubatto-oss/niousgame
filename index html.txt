<body>
  <div class="container">
    <h1>🎮 Game Share Board</h1>

    <div class="form">
      <input id="name" placeholder="Nom du jeu">
      <input id="link" placeholder="Lien (Steam, etc)">
      <input id="price" placeholder="Prix">
      <textarea id="pros" placeholder="Les +"></textarea>
      <textarea id="cons" placeholder="Les -"></textarea>
      <textarea id="review" placeholder="Ton avis"></textarea>
      <button onclick="addGame()">Ajouter</button>
    </div>

    <div id="games"></div>
  </div>

  <script>
    let games = JSON.parse(localStorage.getItem('games') || '[]');

    function save() {
      localStorage.setItem('games', JSON.stringify(games));
    }

    function addGame() {
      const game = {
        name: document.getElementById('name').value,
        link: document.getElementById('link').value,
        price: document.getElementById('price').value,
        pros: document.getElementById('pros').value,
        cons: document.getElementById('cons').value,
        review: document.getElementById('review').value
      };

      if (!game.name) return alert("Nom requis");

      games.push(game);
      save();
      render();
    }

    function render() {
      const container = document.getElementById('games');
      container.innerHTML = '';

      games.forEach((g, index) => {
        const div = document.createElement('div');
        div.className = 'card';

        div.innerHTML = `
          <div class="title">${g.name}</div>
          <div><a href="${g.link}" target="_blank">Lien</a></div>
          <div>💰 ${g.price}</div>
          <div><strong>+ :</strong> ${g.pros}</div>
          <div><strong>- :</strong> ${g.cons}</div>
          <div><strong>Avis :</strong> ${g.review}</div>
          <button onclick="deleteGame(${index})">Supprimer</button>
        `;

        container.appendChild(div);
      });
    }

    function deleteGame(index) {
      games.splice(index, 1);
      save();
      render();
    }

    render();
  </script>
